removeInterval <- function(w, start, end){
  index <- regexpr(start, w, fixed = TRUE)
  while(index[1] != -1){
    if(regexpr(end, w, fixed = TRUE)[1] == -1){
      w <- sub(substr(w, index[1], nchar(w)), "", w, fixed = T)
      break 
      }
    w <- sub(substr(w, index[1], regexpr(end, w, fixed = TRUE)), "", w, fixed = T)
    index <- regexpr(start, w, fixed = TRUE)
  }
  return(w)
}
